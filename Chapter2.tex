	\begin{align}
		&a:(-\otimes-)\otimes-\Rightarrow-\otimes(-\otimes-)\\
		&l:\mathbb{1}\otimes-\Ra\id_\sC & r:-\otimes\mathbb{1}\Ra\id_{sC}
	\end{align}
